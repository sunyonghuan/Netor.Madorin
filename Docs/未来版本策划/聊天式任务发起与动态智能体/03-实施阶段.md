```csharp
    {
        var requestId = Guid.NewGuid().ToString("N");
        publisher.Publish(Events.OnApprovalRequested, new ApprovalRequestArgs(
            TaskId: taskId,
            RequestId: requestId,
            IsPlanApproval: true,
            ProposedAgents: [new DynamicAgentDescriptor(name, instructions, responsibility, requiredTools)]));

        // 等用户决策（复用阶段 5B HITL 等待逻辑）
        var approval = await approvalGate.WaitForDecisionAsync(taskId, requestId);
        if (!approval.IsApproved) return "用户拒绝创建子智能体";
    }

    dynamicRegistry.Register(taskId, new DynamicAgentRecord(...));
    return "已创建子智能体 " + name;
}
```

## 8. 风险预案

### 8.1 中途发现重大设计缺陷

如果在 P2-2 / P2-3 阶段发现 Magentic SDK 不支持动态注入子 agent（理论上 SDK 支持，但实际 AOT 模式下可能有问题）：
- 回退方案 A：把 P2-2 / P2-3 改为"用户在输入框上方多选已有 Agent"（简化为 GroupChat 同款形态），P3 范围不变
- 回退方案 B：暂停 P2，重新评估 SDK 能力 → 若不支持则永久放弃动态创建，仅做 P2-1（聊天式输入）+ P3

### 8.2 用户测试反馈强烈不满

- 单 commit 回退到 P1 之前（所有 P2 改造作为 backlog 重新评估）
- README §6 历史变更记录失败原因 + 经验教训

### 8.3 工期延误

- 若 P2 延误超 1 周（5 周），暂停 P3 启动，先收尾 P2 + 用户实测验证
- 若 P3 延误超 0.5 周（2.5 周），删减 P3-2 文件夹附件（→ 远期 backlog），仅做 P3-1 可视化
